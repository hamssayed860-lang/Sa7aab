            <div class="post-actions">
                <button class="btn-secondary" onclick="alert('قريباً سيتم إضافة رفع الصور!')">📸 صورة/فيديو</button>
                <button class="btn-primary" id="publishBtn">نشر في الفضاء الرقمي ⚡</button>

    <script>
        // قراءة البيانات المحفوظة
        let currentUser = localStorage.getItem('znet_user') || "القائد";
        let totalWalletPoints = parseFloat(localStorage.getItem('znet_wallet')) || 150.00;

        // تحديث المظهر بالاسم الحالي
        function updateUI() {
            document.getElementById('profileHeaderBtn').innerText = currentUser;
            document.getElementById('welcomeMessage').innerText = `انشر فكرتك للعالم يا ${currentUser}..`;
            document.getElementById('globalWallet').innerText = `💰 ${totalWalletPoints.toFixed(2)} نقطة`;
        }

        // تشغيل زر حسابي لتعديل الاسم
        document.getElementById('profileHeaderBtn').addEventListener('click', function() {
            let newName = prompt("ما هو اسمك الجديد لتسجيله في المنصة؟", currentUser);
            if (newName && newName.trim() !== "") {
                currentUser = newName.trim();
                localStorage.setItem('znet_user', currentUser);
                updateUI();
                loadPosts();
            }
        });

        // دالة عرض المنشورات
        function loadPosts() {
            let feed = document.getElementById('postFeed');
            feed.innerHTML = ''; 

            // منشور ترحيبي ثابت
            let defaultPost = document.createElement('div');
            defaultPost.className = 'post';
            defaultPost.innerHTML = `
                <div class="post-header"><div class="avatar">👤</div><div><h4>المستخدم الأول (الشريك الرقمي)</h4><span class="time">منذ دقيقتين</span></div></div>
                <p class="post-content">أهلاً بك في النسخة النهائية المفتوحة! اضغط على زر "حسابي" بالأعلى لتغيير اسمك فوراً، ثم جرب النشر والتفاعل بنجاح وبدون أي قيود.</p>
                <div class="post-footer"><button style="color:#ef4444; font-weight:bold;">❤️ تفاعل (1200)</button><button>💬 نقاش (432)</button><span class="earnings">⚡ أرباح ثابتة: +4.50$</span></div>
            `;
            feed.appendChild(defaultPost);

            // تحميل بقية المنشورات من الذاكرة
            let savedPosts = JSON.parse(localStorage.getItem('znet_posts')) || [];
            savedPosts.forEach((post, index) => {
                let postDiv = document.createElement('div');
                postDiv.className = 'post';
                postDiv.style.marginTop = '20px';
                let likedClass = post.isLiked ? 'liked' : '';

                postDiv.innerHTML = `
                    <div class="post-header">
                        <div class="avatar">👑</div>
                        <div><h4>${post.author}</h4><span class="time">${post.time}</span></div>
                    </div>
                    <p class="post-content">${post.content}</p>
                    <div class="post-footer">
                        <button class="like-btn ${likedClass}" onclick="handleLike(${index})">❤️ تفاعل (<span>${post.likes}</span>)</button>
                        <button>💬 نقاش (0)</button>
                        <span class="earnings">⚡ أرباح المنشور: +${post.earning}$</span>
                    </div>
                `;
                feed.insertBefore(postDiv, feed.firstChild);
            });
        }

        // زر النشر
        document.getElementById('publishBtn').addEventListener('click', function() {
            let textInput = document.getElementById('myInput').value;
            if (textInput.trim() === "") {
                alert("اكتب فكرتك أولاً قبل النشر!");
                return;
            }

            let savedPosts = JSON.parse(localStorage.getItem('znet_posts')) || [];
            let randomEarning = (Math.random() * 5).toFixed(2);

            let newPostData = {
                author: currentUser,
                content: textInput,
                time: new Date().toLocaleTimeString('ar-EG', {hour: '2-digit', minute:'2-digit'}),
                likes: 0,
                isLiked: false,
                earning: randomEarning
            };

            savedPosts.push(newPostData);
            localStorage.setItem('znet_posts', JSON.stringify(savedPosts));
            document.getElementById('myInput').value = "";
            loadPosts();
        });

        // زر التفاعل (اللايك)
        window.handleLike = function(index) {
            let savedPosts = JSON.parse(localStorage.getItem('znet_posts')) || [];
            let post = savedPosts[index];

            if (!post.isLiked) {
                post.isLiked = true; post.likes += 1; totalWalletPoints += 1.50;
            } else {
                post.isLiked = false; post.likes -= 1; totalWalletPoints -= 1.50;
            }

            savedPosts[index] = post;
            localStorage.setItem('znet_posts', JSON.stringify(savedPosts));
            localStorage.setItem('znet_wallet', totalWalletPoints);
            updateUI();
            loadPosts();
        };

        // زر المحفظة
        document.getElementById('walletMenuBtn').addEventListener('click', function() {
            let savedPosts = JSON.parse(localStorage.getItem('znet_posts')) || [];
            alert(`💳 تفاصيل محفظة [ ${currentUser} ]:\n-----------------------\n• رصيدك الإجمالي الحالي: ${totalWalletPoints.toFixed(2)} نقطة.\n• عدد منشوراتك: ${savedPosts.filter(p => p.author === currentUser).length} منشور.`);
        });

        // تشغيل الموقع لأول مرة
        updateUI();
        loadPosts();
    </script>
